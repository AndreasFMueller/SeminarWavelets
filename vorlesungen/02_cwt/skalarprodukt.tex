\begin{frame}
\frametitle{Skalarprodukt}
Abgetastete Signale:
\[
\langle x,y\rangle
=
\sum_{k\in\mathbb Z} x_k\bar{y}_k
\]
$f$, $g$ zeitaghängige Signale $t\mapsto f(t)$ und $t\mapsto g(t)$
\[
\langle f,g\rangle
=
\int_{-\infty}^\infty f(t)\bar{g}(t)\,dt
\]
\end{frame}
