\begin{tikzpicture}[>=latex]

\node at ({1*3.1415},-\s) [below] {$\mathstrut\pi$};
\node at ({2*3.1415},-\s) [below] {$\mathstrut2\pi$};
\node at ({-1*3.1415},-\s) [below] {$\mathstrut-\pi$};
