\def\punkt#1#2{
        \fill[color=white] ({#1},{#2}) circle[radius=0.1];
        \draw[line width=1pt] ({#1},{#2}) circle[radius=0.06];
}

\def\punktblau#1#2{
        \fill[color=white] ({3*#1/2},{3*#2/2}) circle[radius=0.1];
        \draw[line width=1pt,color=blue] ({3*#1/2},{3*#2/2}) circle[radius=0.06];
}
