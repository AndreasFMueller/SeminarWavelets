\def\A{5}
\def\B{1}
\pgfmathparse{2/(\A*\B)}
\xdef\c{\pgfmathresult}

\def\gerade#1#2{

	\xdef\xold{#2}
	\xdef\yold{0}

	\foreach \i in {1,...,15}{
		\xdef\x{\xold}
		\pgfmathparse{\x+(\c)*(1-\x*#1)}
		\xdef\y{\pgfmathresult}

		\draw[line width=1pt,color=gray] ({\xold},{\yold})--({\x},{\y});

		\xdef\xold{\x}
		\xdef\yold{\y}
		\xdef\x{\y}

		\draw[line width=1pt,color=gray] ({\xold},{\yold})--({\x},{\y});

		\xdef\xold{\x}
		\xdef\yold{\y}
	}

	\draw[line width=1.0pt,color=blue]
		(-1,{-1+(\c)*(1+#1)})--(\maxx,{\maxx+(\c)*(1-(\maxx*#1))});

}

\gerade{4}{0.2}
\gerade{3}{0.5}
\gerade{2}{1.5}
\gerade{1}{2.5}

\def\punkt#1{
	\fill[color=red] ({#1},{#1}) circle[radius=0.02];
}
